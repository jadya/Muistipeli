1   package muistipeli.gui;
2   
3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import javax.swing.JButton;
6   
7   /**
8    * Luokka tarjoaa käyttöliittymän valintapainikkeisiin liittyviä metodeita.
9    */
10  public class ValintaPainike extends JButton implements ActionListener {
11  
12      private int numero;
13      private String kohde;
14      private Kayttoliittyma kayttoliittyma;
15      private ValintaPaneeli valintaPaneeli;
16  
17      /**
18       * Konstruktori valintaPainikkeelle.
19       *
20       * @param kayttoliittyma käyttöliittymä, johon valintaPainike liittyy
21       * @param vp valintapaneeli, johon valintapainike lisätään
22       * @param numero valintapainikkeen numero
23       * @param kohde kohde, jonka arvon valitsemiseen valintapainiketta käytetään
24       */
25      public ValintaPainike(Kayttoliittyma kayttoliittyma, ValintaPaneeli vp, int numero, String kohde) {
26          this.numero = numero;
27          this.valintaPaneeli = vp;
28          this.setText("" + numero);
29          this.kohde = kohde;
30          this.kayttoliittyma = kayttoliittyma;
31          this.addActionListener(this);
32      }
33  
34      @Override
35      public void actionPerformed(ActionEvent e) {
36          boolean ok = true;
37          if (this.kohde.equals("korkeus")) {
38              this.kayttoliittyma.getPelinakyma().setKorkeus(numero);
39          }
40          if (this.kohde.equals("leveys")) {
41              if ((this.kayttoliittyma.getPelinakyma().getKorkeus() * numero) % 2 == 0) {
42                  this.kayttoliittyma.getPelinakyma().setLeveys(numero);
43              } else {
44                  this.kayttoliittyma.getPelinakyma().setLeveys(numero + 1);
45                  this.valintaPaneeli.vahvistaValinta(numero + 1);
46                  ok = false;
47              }
48          }
49          if (this.kohde.equals("pelaajien maara")) {
50              this.kayttoliittyma.getPelinakyma().setPelaajienMaara(numero);
51          }
52          if (this.kohde.equals("tekoalyjen maara")) {
53              if (this.kayttoliittyma.getPelinakyma().getPelaajienMaara() >= numero) {
54                  this.kayttoliittyma.getPelinakyma().setTekoalyjenMaara(numero);
55              } else {
56                  ok = false;
57              }
58          }
59          if (ok) {
60              this.valintaPaneeli.vahvistaValinta(this.numero);
61          }
62      }
63  
64      public int getNumero() {
65          return this.numero;
66      }
67  
68  }
