3   import java.awt.event.ActionEvent;
4   import java.awt.event.ActionListener;
5   import javax.swing.JButton;
6   
7   public class ValintaPainike extends JButton implements ActionListener {
8   
9       private int numero;
10      private String kohde;
11      private Kayttoliittyma kayttoliittyma;
12  
13      public ValintaPainike(Kayttoliittyma kayttoliittyma, int numero, String kohde) {
14          this.numero = numero;
15          this.setText("" + numero);
16          this.kohde = kohde;
17          this.kayttoliittyma = kayttoliittyma;
18          this.addActionListener(this);
19      }
20  
21      @Override
22      public void actionPerformed(ActionEvent e) {
23          if (this.kohde.equals("korkeus")) {
24              this.kayttoliittyma.getPelinakyma().setKorkeus(numero);
25          }
26          if (this.kohde.equals("leveys")) {
27              this.kayttoliittyma.getPelinakyma().setLeveys(numero);
28          }
29          if (this.kohde.equals("pelaajien maara")) {
30              this.kayttoliittyma.getPelinakyma().setPelaajienMaara(numero);
31          }
32      }
