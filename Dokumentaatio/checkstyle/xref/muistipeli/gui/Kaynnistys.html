1   package muistipeli.gui;
2   
3   import java.awt.Component;
4   import java.awt.event.ActionEvent;
5   import java.awt.event.ActionListener;
6   import java.awt.event.WindowEvent;
7   import java.io.IOException;
8   import java.util.Collections;
9   import javax.swing.JButton;
10  import javax.swing.JOptionPane;
11  
12  /**
13   * Luokka tarjoaa käyttöliittymän käynnistyspainikkeeseen liittyviä metodeita.
14   */
15  public class Kaynnistys extends JButton implements ActionListener {
16  
17      private String toiminto;
18      private final Kayttoliittyma kayttoliittyma;
19  
20      /**
21       * Konstruktori käynnistykselle.
22       *
23       * @param kayttoliittyma käyttöliittymä, johon käynnistys-painike liittyy
24       * @param toiminto toiminto, joka toteutetaan käynnistys-painiketta
25       * klikatessa
26       */
27      public Kaynnistys(Kayttoliittyma kayttoliittyma, String toiminto) {
28          this.toiminto = toiminto;
29          this.setText(toiminto);
30          this.kayttoliittyma = kayttoliittyma;
31          this.addActionListener(this);
32      }
33  
34      @Override
35      public void actionPerformed(ActionEvent e) {
36          if (this.toiminto.equals("Lopeta peli")) {
37              this.kayttoliittyma.getPeli().lopetaPeli();
38              Collections.sort(this.kayttoliittyma.getPeli().getPelaajat());
39              this.kayttoliittyma.getNakymanVaihtaja().siirryPistenakymaan();
40          }
41          if (this.toiminto.equals("Aloita peli")) {
42              try {
43                  this.kayttoliittyma.getNakymanVaihtaja().siirryPelinakymaan();
44              } catch (IOException ex) {
45                  JOptionPane.showMessageDialog((Component) e.getSource(), "Jotain meni nyt pahasti pieleen.");
46                  this.kayttoliittyma.getFrame().dispatchEvent(new WindowEvent(this.kayttoliittyma.getFrame(), WindowEvent.WINDOW_CLOSING));
47              }
48              this.setToiminto("Lopeta peli");
49          }
50          if (this.toiminto.equals("Uusi peli")) {
51              try {
52                  this.kayttoliittyma.getNakymanVaihtaja().siirryValikkoon();
53              } catch (IOException ex) {
54                  JOptionPane.showMessageDialog((Component) e.getSource(), "Jotain meni nyt pahasti pieleen.");
55                  this.kayttoliittyma.getFrame().dispatchEvent(new WindowEvent(this.kayttoliittyma.getFrame(), WindowEvent.WINDOW_CLOSING));
56              }
57              this.setToiminto("Aloita peli");
58          }
59      }
60  
61      public String getToiminto() {
62          return this.toiminto;
63      }
64  
65      /**
66       * Metodi asettaa käynnistys-painikkeelle toiminnon ja muuttaa painikkeen
67       * tekstin sitä vastaavaksi.
68       *
69       * @param toiminto toiminto, joka käynnistys-painikkeeseen asetetaan
70       */
71      public void setToiminto(String toiminto) {
72          this.toiminto = toiminto;
73          this.setText(toiminto);
74      }
75  
76  }
