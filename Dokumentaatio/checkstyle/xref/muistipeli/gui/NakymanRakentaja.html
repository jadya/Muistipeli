3   import java.awt.BorderLayout;
4   import java.awt.Container;
5   import java.awt.GridLayout;
6   import java.io.IOException;
7   import javax.swing.JLabel;
8   import javax.swing.JPanel;
9   
10  /**
11   * Luokka tarjoaa käyttöliittymän näkymien rakentamiseen liittyviä metodeita.
12   */
13  public class NakymanRakentaja {
14  
15      private final Kayttoliittyma kayttoliittyma;
16  
17      /**
18       * Konstruktori nakymanRakentajalle.
19       *
20       * @param kl käyttöliittymä, johon näkymän rakentaja liittyy
21       */
22      public NakymanRakentaja(Kayttoliittyma kl) {
23          this.kayttoliittyma = kl;
24      }
25  
26      /**
27       * Metodi pelinäkymän rakentamiseen. Lisää pelinäkymän komponentit annettuun
28       * säiliöön.
29       *
30       * @param leveys pelinäkymän pelin pelialustan leveys
31       * @param korkeus pelinäkymän pelin pelialustan korkeus
32       * @param c säiliö, johon luodut komponentit lisätään
33       * @throws java.io.IOException keskeytys
34       */
35      public void rakennaPelinakyma(int leveys, int korkeus, Container c) throws IOException {
36          this.kayttoliittyma.setTekoalyllaVuoroKesken(false);
37          for (int i = 0; i < c.getComponentCount(); i++) {
38              c.remove(i);
39          }
40          this.kayttoliittyma.setVuoronNayttaja(new VuoronNayttaja(this.kayttoliittyma.getPeli()));
41          c.add(this.kayttoliittyma.getVuoronNayttaja(), BorderLayout.NORTH);
42          c.add(new Kaynnistys(this.kayttoliittyma, "Lopeta peli"), BorderLayout.SOUTH);
43          JPanel panel = new JPanel();
44          panel.setLayout(new GridLayout(korkeus, leveys));
45          for (int i = 0; i < korkeus; i++) {
46              for (int j = 0; j < leveys; j++) {
47                  Korttipaikka korttipaikka = new Korttipaikka(this.kayttoliittyma, this.kayttoliittyma.getPeli().getPelialusta().getKortti(j, i), this.kayttoliittyma.getPeli().getPelialusta());
48                  this.kayttoliittyma.getKorttipaikat().add(korttipaikka);
49                  panel.add(korttipaikka);
50              }
51          }
52          c.add(panel, BorderLayout.CENTER);
53      }
54  
55      /**
56       * Metodi valikon rakentamiseen. Lisää valikon komponentit annettuun
57       * säiliöön.
58       *
59       * @param c säiliö, johon luodut komponentit lisätään
60       * @throws java.io.IOException keskeytys
61       */
62      public void rakennaValikko(Container c) throws IOException {
63          for (int i = 0; i < c.getComponentCount(); i++) {
64              c.remove(i);
65          }
66          this.kayttoliittyma.setPelinakyma(new Pelinakyma());
67          c.add(new JLabel("Muistipeli"), BorderLayout.NORTH);
68          c.add(new Kaynnistys(this.kayttoliittyma, "Aloita peli"), BorderLayout.SOUTH);
69          JPanel panel = new JPanel();
70          panel.setLayout(new GridLayout(8, 1));
71          panel.add(new JLabel("Valitse pelialustan korkeus:"));
72          panel.add(new ValintaPaneeli(this.kayttoliittyma, "korkeus", 2, 10));
73          panel.add(new JLabel("Valitse pelialustan leveys:"));
74          panel.add(new ValintaPaneeli(this.kayttoliittyma, "leveys", 2, 10));
75          panel.add(new JLabel("Valitse pelaajien maara:"));
76          panel.add(new ValintaPaneeli(this.kayttoliittyma, "pelaajien maara", 2, 10));
77          panel.add(new JLabel("Tekoalyjen maara pelaajista:"));
78          panel.add(new ValintaPaneeli(this.kayttoliittyma, "tekoalyjen maara", 0, 10));
79          c.add(panel, BorderLayout.CENTER);
80      }
81  
82      /**
83       * Metodi pistenäkymän rakentamiseen. Lisää pistenäkymän komponentit
84       * annettuun säiliöön.
85       *
86       * @param c säiliö, johon luodut komponentit lisätään
87       */
88      public void rakennaPistenakyma(Container c) {
89          for (int i = 0; i < c.getComponentCount(); i++) {
90              c.remove(i);
91          }
92          c.add(new JLabel("Muistipeli"), BorderLayout.NORTH);
93          c.add(new Kaynnistys(this.kayttoliittyma, "Uusi peli"), BorderLayout.SOUTH);
94          JPanel panel = new JPanel();
95          panel.setLayout(new GridLayout(3 + this.kayttoliittyma.getPeli().getPelaajat().size() + this.kayttoliittyma.getPeli().johdossa().size(), 1));
96          panel.add(new JLabel("Peli ohi"));
97          if (this.kayttoliittyma.getPeli().johdossa().size() > 1) {
98              panel.add(new JLabel("Voittajat: "));
99              for (int i = 0; i < kayttoliittyma.getPeli().johdossa().size(); i++) {
100                 panel.add(new JLabel(kayttoliittyma.getPeli().johdossa().get(i).toString()));
101             }
102         } else if (this.kayttoliittyma.getPeli().johdossa().size() == 1) {
103             panel.add(new JLabel("Voittaja: "));
104             panel.add(new JLabel(kayttoliittyma.getPeli().johdossa().get(0).toString()));
105         }
106         panel.add(new JLabel("Tulokset:"));
107         this.kayttoliittyma.getPeli().getPelaajat().stream().forEach((p) -> {
108             panel.add(new JLabel(p.toString()));
109         });
110         c.add(panel, BorderLayout.CENTER);
111     }

113 }
