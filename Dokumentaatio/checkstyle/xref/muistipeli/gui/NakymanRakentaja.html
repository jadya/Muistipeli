2   package muistipeli.gui;
3   
4   import java.awt.BorderLayout;
5   import java.awt.Container;
6   import java.awt.GridLayout;
7   import javax.swing.JLabel;
8   import javax.swing.JPanel;
9   
10  /**
11   * Luokka tarjoaa käyttöliittymän näkymien rakentamiseen liittyviä metodeita.
12   */
13  public class NakymanRakentaja {
14      
15      private final Kayttoliittyma kayttoliittyma;
16      
17      /**
18       * Konstruktori nakymanRakentajalle.
19       * @param kl käyttöliittymä, johon näkymän rakentaja liittyy
20       */
21      public NakymanRakentaja(Kayttoliittyma kl) {
22          this.kayttoliittyma = kl;
23      }
24      
25      /**
26       * Metodi pelinäkymän rakentamiseen. Lisää pelinäkymän komponentit annettuun
27       * säiliöön.
28       * @param leveys pelinäkymän pelin pelialustan leveys
29       * @param korkeus pelinäkymän pelin pelialustan korkeus
30       * @param c säiliö, johon luodut komponentit lisätään
31       */
32      public void rakennaPelinakyma(int leveys, int korkeus, Container c) {
33          this.kayttoliittyma.setTekoalyllaVuoroKesken(false);
34          for (int i = 0; i < c.getComponentCount(); i++) {
35              c.remove(i);
36          }
37          this.kayttoliittyma.setVuoronNayttaja(new VuoronNayttaja(this.kayttoliittyma.getPeli()));
38          c.add(this.kayttoliittyma.getVuoronNayttaja(), BorderLayout.NORTH);
39          c.add(new Kaynnistys(this.kayttoliittyma, "Lopeta peli"), BorderLayout.SOUTH);
40          JPanel panel = new JPanel();
41          panel.setLayout(new GridLayout(korkeus, leveys));
42          for (int i = 0; i < korkeus; i++) {
43              for (int j = 0; j < leveys; j++) {
44                  Korttipaikka korttipaikka = new Korttipaikka(this.kayttoliittyma, this.kayttoliittyma.getPeli().getPelialusta().getKortti(j, i), this.kayttoliittyma.getPeli().getPelialusta());
45                  this.kayttoliittyma.getKorttipaikat().add(korttipaikka);
46                  panel.add(korttipaikka);
47              }
48          }
49          c.add(panel, BorderLayout.CENTER);
50      }
51      
52      /**
53       * Metodi valikon rakentamiseen. Lisää valikon komponentit annettuun
54       * säiliöön.
55       * @param c säiliö, johon luodut komponentit lisätään
56       */
57      public void rakennaValikko(Container c) {
58          for (int i = 0; i < c.getComponentCount(); i++) {
59              c.remove(i);
60          }
61          this.kayttoliittyma.setPelinakyma(new Pelinakyma());
62          c.add(new JLabel("Muistipeli"), BorderLayout.NORTH);
63          c.add(new Kaynnistys(this.kayttoliittyma, "Aloita peli"), BorderLayout.SOUTH);
64          JPanel panel = new JPanel();
65          panel.setLayout(new GridLayout(8, 1));
66          panel.add(new JLabel("Valitse pelialustan korkeus:"));
67          panel.add(new ValintaPaneeli(this.kayttoliittyma, "korkeus", 2, 10));
68          panel.add(new JLabel("Valitse pelialustan leveys:"));
69          panel.add(new ValintaPaneeli(this.kayttoliittyma, "leveys", 2, 10));
70          panel.add(new JLabel("Valitse pelaajien maara:"));
71          panel.add(new ValintaPaneeli(this.kayttoliittyma, "pelaajien maara", 2, 10));
72          panel.add(new JLabel("Tekoalyjen maara pelaajista:"));
73          panel.add(new ValintaPaneeli(this.kayttoliittyma, "tekoalyjen maara", 0, 1));
74          c.add(panel, BorderLayout.CENTER);
75      }
76      
77      /**
78       * Metodi pistenäkymän rakentamiseen. Lisää pistenäkymän komponentit annettuun
79       * säiliöön.
80       * @param c säiliö, johon luodut komponentit lisätään
81       */
82      public void rakennaPistenakyma(Container c) {
83          for (int i = 0; i < c.getComponentCount(); i++) {
84              c.remove(i);
85          }
86          c.add(new JLabel("Muistipeli"), BorderLayout.NORTH);
87          c.add(new Kaynnistys(this.kayttoliittyma, "Uusi peli"), BorderLayout.SOUTH);
88          JPanel panel = new JPanel();
89          panel.setLayout(new GridLayout(3 + this.kayttoliittyma.getPeli().getPelaajat().size() + this.kayttoliittyma.getPeli().johdossa().size(), 1));
90          panel.add(new JLabel("Peli ohi"));
91          if (this.kayttoliittyma.getPeli().johdossa().size() > 1) {
92              panel.add(new JLabel("Voittajat: "));
93              for (int i = 0; i < kayttoliittyma.getPeli().johdossa().size(); i++) {
94                  panel.add(new JLabel(kayttoliittyma.getPeli().johdossa().get(i).toString()));
95              }
96          } else if (this.kayttoliittyma.getPeli().johdossa().size() == 1) {
97              panel.add(new JLabel("Voittaja: "));
98              panel.add(new JLabel(kayttoliittyma.getPeli().johdossa().get(0).toString()));
99          }
100         panel.add(new JLabel("Tulokset:"));
101         this.kayttoliittyma.getPeli().getPelaajat().stream().forEach((p) -> {
102             panel.add(new JLabel(p.toString()));
103         });
104         c.add(panel, BorderLayout.CENTER);
105     }
106     
107 }
